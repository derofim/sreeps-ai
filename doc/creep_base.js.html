/**
 * Efficiently moves creep at target.
 *
 * @param {Object} creep - Current creep
 * @return Success of action.
 */
Creep.prototype.moveAt = function(target) {
	if (this.fatigue > 0) {
		return this;
	}

	this.moveTo(target);

	return this;
};
